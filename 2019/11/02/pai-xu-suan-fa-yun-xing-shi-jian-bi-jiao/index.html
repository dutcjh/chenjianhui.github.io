冒泡排序

def bubbleSort(arr):
    for i in range(1, len(arr)):
        flag = False  # 停止标志
        for j in range(len(arr)-i):
            if arr[j] > arr[j+1]:
                arr[j], arr[j+1] = arr[j+1], arr[j]
                flag = True
        if not flag:break
    return arr

选择排序：

def selectionSort(arr):
    for i in range(len(arr)-1):
        minIndex = i;  # 记录最小数的索引
        for j in range(i+1,len(arr)):
            if arr[j] < arr[minIndex]:
                minIndex = j;  # i不是最小数时，将 i和最小数进行交换
        if i != minIndex:
            arr[i], arr[minIndex] = arr[minIndex], arr[i]
    return arr

def insertionSort(arr, left = None, right = None):
    left = 0 if not isinstance(left, (int, float)) else left
    right = len(arr)-1 if not isinstance(right, (int, float)) else right
    for i in range(left, right+1):
        preIndex = i-1
        current = arr[i]
        while preIndex >= left and arr[preIndex] > current:
            arr[preIndex+1] = arr[preIndex]
            preIndex = preIndex-1
        arr[preIndex+1] = current
    return arr

归并排序

def mergeSort(arr):
    #不断递归调用自己一直到拆分成成单个元素的时候就返回这个元素，不再拆分了
    if len(arr) < 2:
        return arr
    mid = len(arr) // 2 #取拆分的中间位置
    #拆分过后左右两侧子串
    left, right = arr[:mid], arr[mid:]
    #对拆分过后的左右再拆分 一直到只有一个元素为止
    #最后一次递归时候ll和lr都会接到一个元素的列表
    # 最后一次递归之前的ll和rl会接收到排好序的子序列
    # 我们对返回的两个拆分结果进行排序后合并再返回正确顺序的子列表
    # 这里我们调用拎一个函数帮助我们按顺序合并ll和lr
    return merge(mergeSort(left), mergeSort(right))

#这里接收两个列表
def merge(left, right):
    # 从两个有顺序的列表里边依次取数据比较后放入result
    # 每次我们分别拿出两个列表中最小的数比较，把较小的放入result
    result = []
    while len(left)>0 and len(right)>0 :
        #为了保持稳定性，当遇到相等的时候优先把左侧的数放进结果列表，因为left本来也是大数列中比较靠左的
        if left[0] <= right[0]:
            result.append( left.pop(0) )
        else:
            result.append( right.pop(0) )
    #while循环出来之后 说明其中一个数组没有数据了，我们把另一个数组添加到结果数组后面
    result += left
    result += right
    return result

快速排序(未改进)

def QuickSort(arr, left = None, right = None): # 传统快排
    left = 0 if not isinstance(left, (int, float)) else left
    right = len(arr)-1 if not isinstance(right, (int, float)) else right
    #判断low是否小于high,如果为false,直接返回
    if left < right:
        i,j = left,right
        #设置基准数
        base = arr[i]

        while i < j:
            #如果列表后边的数,比基准数大或相等,则前移一位直到有比基准数小的数出现
            while (i < j) and (arr[j] >= base):
                j = j - 1

            #如找到,则把第j个元素赋值给第个元素i,此时表中i,j个元素相等
            arr[i] = arr[j]

            #同样的方式比较前半区
            while (i < j) and (arr[i] <= base):
                i = i + 1
            arr[j] = arr[i]
        #做完第一轮比较之后,列表被分成了两个半区,并且i=j,需要将这个数设置回base
        arr[i] = base

        #递归前后半区
        quickSort(arr, left, i - 1)
        quickSort(arr, j + 1, right)
    return arr

def insertionSort(arr, left = None, right = None):
    left = 0 if not isinstance(left, (int, float)) else left
    right = len(arr)-1 if not isinstance(right, (int, float)) else right
    for i in range(left, right+1):

        current = arr[i]
        while preIndex >= left and arr[preIndex] > current:
            arr[preIndex+1] = arr[preIndex]
            preIndex = preIndex-1
        arr[preIndex+1] = current
    return arr

def quickSort(arr, left = None, right = None):
    left = 0 if not isinstance(left, (int, float)) else left
    right = len(arr)-1 if not isinstance(right, (int, float)) else right
    length = right - left
    if length > 10:
        partitionIndex = partition(arr, left, right)
        quickSort(arr, left, partitionIndex)
        quickSort(arr, partitionIndex+1, right)
    else:
        insertionSort(arr, left, right)
        return arr

def partition(arr, left, right):
    # 把中间大的数放在最左边作为基准值
    medium = (left+right)//2
    if arr[left] > arr[right]:
        arr[left], arr[right] = arr[right], arr[left]
    if arr[medium] > arr[right]:
        arr[medium], arr[right] = arr[right], arr[medium]
    if arr[medium] > arr[left]:
        arr[left], arr[medium] = arr[medium], arr[left]
    # 基准
    pivot = arr[left]
    i,j = left,right
    while i < j:
        #如果列表后边的数,比基准数大或相等,则前移一位直到有比基准数小的数出现
        while (i < j) and (arr[j] >= pivot):
            j = j - 1
        #如找到,则把第j个元素赋值给第个元素i,此时表中i,j个元素相等
        arr[i] = arr[j]

        #同样的方式比较前半区
        while (i < j) and (arr[i] <= pivot):
            i = i + 1
        arr[j] = arr[i]
        #做完第一轮比较之后,列表被分成了两个半区,并且i=j,需要将这个数设置回pivot
    arr[i] = pivot
    #做完第一轮比较之后,列表被分成了两个半区,并且i=j,需要将这个数返回
    return i
